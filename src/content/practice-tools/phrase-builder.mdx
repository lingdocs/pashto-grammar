import NPPicker from "../../components/np-picker/NPPicker";
import { useState } from "react";

export function Display() {
    const [np, setNp] = useState(undefined);
    return <div>
        <NPPicker np={np} onChange={setNp} />
        <pre>
            {JSON.stringify(np, null, "  ")}
        </pre>
    </div>
}

<Display />
